class Solution {
public:
    int maxArea(vector<int>& height) {
        int n = height.size();
        int i = 0, j = n - 1;
        int m = -1;
        while (i < j){
            int res = (j - i) * min(height[i], height[j]);
            m = max(res, m);
            if (height[i] <= height[j]) i ++ ;
            else j -- ;
        }
        return m;
    }
};
